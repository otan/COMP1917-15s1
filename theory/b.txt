7. Find two potential flaws with this code, and attempt to fix them. (4 marks)

#include <stdio.h>
#include <stdlib.h>

#define MAX_LENGTH 20
#define BUFFER_LENGTH MAX_LENGTH + 1

int main(void) {
    char s[BUFFER_LENGTH];
    char c = getchar();
    int i = 0;
    while (c != EOF && i < BUFFER_LENGTH) {
        s[i] = c;
        i++;
        c = getchar();
    }
    s[i] = ‘\0’;
    printf(“%s\n”, s);
    return 0;
}
